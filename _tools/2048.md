      <span class="game-stat">Best<b id="g8-best">0</b></span>

<script>
(function(){
  var boardEl=document.getElementById('g8-board'),scoreEl=document.getElementById('g8-score'),bestEl=document.getElementById('g8-best'),statusEl=document.getElementById('g8-status');
  var size=4,grid,score=0,best=0,won=false,over=false;
  try{best=parseInt(localStorage.getItem('tv_g8_best')||'0',10)||0;}catch(e){}
  bestEl.textContent=best;
  var colors={2:'#3b3b3b',4:'#4b3f2f',8:'#7c4a1e',16:'#a8531a',32:'#c2410c',64:'#ea580c',128:'#ca8a04',256:'#a16207',512:'#854d0e',1024:'#65a30d',2048:'#16a34a'};
  function empty(){grid=[];for(var y=0;y<size;y++){grid[y]=[];for(var x=0;x<size;x++)grid[y][x]=0;}}
  function addTile(){
    var cells=[];
    for(var y=0;y<size;y++)for(var x=0;x<size;x++)if(grid[y][x]===0)cells.push([x,y]);
    if(!cells.length)return;
    var c=cells[Math.floor(Math.random()*cells.length)];
    grid[c[1]][c[0]]=Math.random()<0.9?2:4;
  }
  function paint(){
    boardEl.innerHTML='';
    for(var y=0;y<size;y++)for(var x=0;x<size;x++){
      var d=document.createElement('div');d.className='retro-cell';d.style.height='78px';d.style.fontSize='1.6rem';
      var v=grid[y][x];
      if(v){d.textContent=v;d.style.background=colors[v]||'#16a34a';d.style.color=v>4?'#fff':'var(--text)';}
      boardEl.appendChild(d);
    }
  }
  function slide(row){
    var a=row.filter(function(v){return v;}),pts=0;
    for(var i=0;i<a.length-1;i++){if(a[i]===a[i+1]){a[i]*=2;pts+=a[i];a.splice(i+1,1);}}
    while(a.length<size)a.push(0);
    return{row:a,pts:pts};
  }
  function move(dir){
    if(over)return;var moved=false,pts=0;
    var g=grid.map(function(r){return r.slice();});
    for(var i=0;i<size;i++){
      var line=[];
      for(var j=0;j<size;j++)line.push(dir==='left'?grid[i][j]:dir==='right'?grid[i][size-1-j]:dir==='up'?grid[j][i]:grid[size-1-j][i]);
      var res=slide(line);line=res.row;pts+=res.pts;
      for(var j=0;j<size;j++){var v=line[j];if(dir==='left')grid[i][j]=v;else if(dir==='right')grid[i][size-1-j]=v;else if(dir==='up')grid[j][i]=v;else grid[size-1-j][i]=v;}
    }
    for(var y=0;y<size;y++)for(var x=0;x<size;x++)if(grid[y][x]!==g[y][x])moved=true;
    if(!moved)return;
    score+=pts;scoreEl.textContent=score;
    if(score>best){best=score;bestEl.textContent=best;try{localStorage.setItem('tv_g8_best',best);}catch(e){}}
    addTile();paint();check();
  }
  function check(){
    var full=true;
    for(var y=0;y<size;y++)for(var x=0;x<size;x++){if(grid[y][x]===0)full=false;if(grid[y][x]===2048&&!won){won=true;statusEl.textContent='You reached 2048! 🎉 Keep going?';}}
    if(full){
      var stuck=true;
      for(var y=0;y<size;y++)for(var x=0;x<size;x++){if(x<size-1&&grid[y][x]===grid[y][x+1])stuck=false;if(y<size-1&&grid[y][x]===grid[y+1][x])stuck=false;}
      if(stuck){over=true;statusEl.textContent='No moves left — Game over.';}
    }
  }
  window.g8New=function(){empty();score=0;won=false;over=false;scoreEl.textContent=0;statusEl.textContent='Combine tiles to reach 2048.';addTile();addTile();paint();};
  document.addEventListener('keydown',function(e){
    var k=e.key.toLowerCase();
    if(k==='arrowleft'||k==='a'){move('left');e.preventDefault();}
    else if(k==='arrowright'||k==='d'){move('right');e.preventDefault();}
    else if(k==='arrowup'||k==='w'){move('up');e.preventDefault();}
    else if(k==='arrowdown'||k==='s'){move('down');e.preventDefault();}
  });
  var tx=0,ty=0;
  boardEl.addEventListener('touchstart',function(e){tx=e.touches[0].clientX;ty=e.touches[0].clientY;},{passive:true});
  boardEl.addEventListener('touchend',function(e){var dx=e.changedTouches[0].clientX-tx,dy=e.changedTouches[0].clientY-ty;if(Math.abs(dx)>Math.abs(dy))move(dx>0?'right':'left');else move(dy>0?'down':'up');},{passive:true});
  g8New();
})();
</script>
